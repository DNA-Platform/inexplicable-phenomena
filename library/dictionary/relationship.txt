:: # Semantic Reference Theory
:: - A Foundational Sketch

:: Definition of the ternary operator
def x =t> y:
    R(x,y,t)

:: Defintion of inheritance
def s -> t: E[x,y]: x =s> y =|> x =t> y
def s -(x,y)> t: x =s> y =|> x =t> y

---

:: The subject-object relationship
E[subject, object, relationship]: 
    subject =relationship> object
    relationship == (subject, object)

:: The reference of a referent
E[reference, referent]: 
    reference =reference> referent
    reference == (reference, referent)
    reference -> relationship

:: The representative aspects of a representation
E[representative, representation]: 
    representative =representation> representation
    representation == (representative, representation)
    representative -> representation -> reference

:: The identification of an identity
E[identification, identity]: 
    identity =identification> identity 

:: The individuality of an individual
E[individuality]: 
    individual =individual> individual
    individual == (individual, individual)
    individual -> identity

:: The subject of dichotomy
E[dichotomy, canonical, indexical]:
    subject: { 
        dichotomy,
        canonical, 
        indexical, 
    }: [
        canonical =dichotomy> indexical
        subject:{ t } =|> canonical:{ t } || indexical:{ t }
    ]

:: The fundamental dichotomy of SRT
E[relational]:
    subject: {
        relationship,
        identification,
    }: [
        relational -> dichotomy
        relationship -> canonical
        identification -> indexical
        relationship =relational:dichotomy> identification
    ]

---

:: The reference conundrum
:: The representation refers to the referent through its representative for it
:: We can infer that the representative is a representation of the referent
:: We can infer that the representation is representative of the referent
:: Let us suppose that the most specific way the reprsentative relates to the referent is through the representation
:: One perspective is that a reference refers to a referent because it is a type of representative in a representation
:: Another perspective is that the reference refers to the referent in the most fundamental way
:: It will be the subject of the reference frame that determines which of these is taken as the fundamental perspective 
representative -> representation -> reference
representation =representative> referent
representative =representation> referent
representative =representation> referent
(representative, referent) == representation
